**技术实现**:
```typescript
interface CodeSnippet {
  id: string;
  code: string;
  language: string;
  
  // 元数据
  title: string;
  description: string;
  tags: string[];
  
  // 来源
  source: {
    conversationId: string;
    messageId: string;
    tool: 'claude-code' | 'aider' | 'continue-dev';
    timestamp: number;
  };
  
  // 使用追踪
  usage: {
    count: number;
    lastUsed: number;
    copiedTo: string[]; // 应用到的项目路径
  };
  
  // 用于智能推荐
  embedding: Float32Array;
  astFingerprint?: string; // 代码结构指纹
}

// 智能推荐算法
class CodeRecommender {
  async recommendForCurrentFile(
    filePath: string,
    fileContent: string,
    cursorPosition: number
  ): Promise<CodeSnippet[]> {
    // 1. 获取当前文件上下文
    const context = this.extractContext(fileContent, cursorPosition);
    
    // 2. 向量化搜索
    const candidates = await this.vectorSearch(context.embedding);
    
    // 3. 基于使用历史加权
    const weighted = candidates.map(c => ({
      ...c,
      score: c.similarity * (1 + Math.log(c.usage.count + 1))
    }));
    
    // 4. 过滤已使用的项目
    const currentProject = getCurrentProject();
    const filtered = weighted.filter(c => 
      !c.usage.copiedTo.includes(currentProject)
    );
    
    return filtered.sort((a, b) => b.score - a.score).slice(0, 5);
  }
}
```

## 5. 技术实现方案

### 5.1 核心算法

#### 代码相似度计算 (用于代码去重和推荐)

```typescript
class CodeSimilarityEngine {
  // 多维度相似度计算
  async calculateSimilarity(a: CodeSnippet, b: CodeSnippet): Promise<number> {
    const weights = {
      semantic: 0.4,    // 语义相似度 (embedding)
      structural: 0.3,  // 结构相似度 (AST)
      token: 0.2,       // Token相似度
      metadata: 0.1     // 元数据相似度 (language, imports)
    };

    const scores = await Promise.all([
      this.semanticSimilarity(a, b),
      this.structuralSimilarity(a, b),
      this.tokenSimilarity(a, b),
      this.metadataSimilarity(a, b)
    ]);

    return Object.entries(weights).reduce((sum, [key, weight], i) => 
      sum + scores[i] * weight, 0);
  }

  // 结构相似度 (基于AST)
  private async structuralSimilarity(a: CodeSnippet, b: CodeSnippet): Promise<number> {
    // 提取AST指纹
    const fingerprintA = this.extractASTFingerprint(a.code);
    const fingerprintB = this.extractASTFingerprint(b.code);
    
    // 计算树编辑距离
    return this.treeEditDistanceSimilarity(fingerprintA, fingerprintB);
  }

  // 语义相似度 (基于embedding)
  private async semanticSimilarity(a: CodeSnippet, b: CodeSnippet): Promise<number> {
    if (!a.embedding || !b.embedding) {
      // 动态生成embedding
      a.embedding = await this.generateEmbedding(a.code);
      b.embedding = await this.generateEmbedding(b.code);
    }
    
    return this.cosineSimilarity(a.embedding, b.embedding);
  }
}
```

#### 决策提取算法

```typescript
class DecisionExtractor {
  async extractDecisions(messages: TerminalMessage[]): Promise<Decision[]> {
    const decisions: Decision[] = [];
    
    // 模式1: 显式决策标记
    const explicitPattern = /(?:decided|decision|conclusion|resolved)\s*:\s*(.+?)(?:\.|\n|$)/gi;
    
    // 模式2: 选择语句
    const choicePattern = /(?:chose|opted for|went with|selected)\s+(?:to\s+)?(.+?)(?:\.|\n|$)/gi;
    
    // 模式3: 建议采纳
    const adoptionPattern = /(?:adopt|implement|use)\s+(?:the\s+)?(.+?)(?:approach|strategy|solution)?/gi;
    
    for (const msg of messages) {
      if (msg.role !== 'ai') continue;
      
      const content = msg.textContent + ' ' + (msg.thinking || '');
      
      // 提取所有模式
      for (const pattern of [explicitPattern, choicePattern, adoptionPattern]) {
        let match;
        while ((match = pattern.exec(content)) !== null) {
          const decision: Decision = {
            id: generateUUID(),
            decision: match[1].trim(),
            rationale: this.extractRationale(content, match.index),
            alternatives: this.extractAlternatives(content),
            timestamp: msg.timestamp,
            messageId: msg.id,
            confidence: this.calculateConfidence(match, content)
          };
          
          decisions.push(decision);
        }
      }
    }
    
    // 去重和排序
    return this.deduplicateAndRank(decisions);
  }
}
```

// 增量处理和懒加载
class PerformanceOptimizer {
  // 虚拟列表用于大量会话
  virtualListConfig = {
    itemHeight: 80,
    overscan: 5,
    maxItems: 1000
  };

  // 增量搜索
  async incrementalSearch(
    query: string,
    onProgress: (results: SearchResult[]) => void
  ): Promise<void> {
    // 1. 先搜索缓存 (100ms)
    const cached = await this.searchCache(query);
    onProgress(cached);
    
    // 2. 后台搜索数据库
    const dbResults = await this.searchDatabase(query);
    onProgress([...cached, ...dbResults]);
    
    // 3. 语义搜索 (较慢)
    const semanticResults = await this.semanticSearch(query);
    onProgress([...cached, ...dbResults, ...semanticResults]);
  }

  // 智能预加载
  async preloadPredictedContent(): Promise<void> {
    const predictions = await this.predictUserNeeds();
    
    for (const prediction of predictions) {
      if (prediction.confidence > 0.7) {
        await this.prefetch(prediction.resource);
      }
    }
  }
}
